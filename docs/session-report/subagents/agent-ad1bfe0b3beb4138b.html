255		// 1. Exact absolute-path match.
256		for i, g := range groups {
257			if g.ProjectPath == want {
258				return i, nil
259			}
260		}
261		// 2. Case-insensitive basename match.
262		var matches []int
263		for i, g := range groups {
264			if strings.EqualFold(filepath.Base(g.ProjectPath), want) {
265				matches = append(matches, i)
266			}
267		}
268		// 3. Fall back to a case-insensitive substring of the full path.
269		if len(matches) == 0 {

271			for i, g := range groups {
272				if strings.Contains(strings.ToLower(g.ProjectPath), lw) {
273					matches = append(matches, i)
274				}
275			}
276		}
277		switch len(matches) {
278		case 0:
279			return -1, fmt.Errorf("no project matching %q", want)
280		case 1:
281			return matches[0], nil
282		default:
283			return -1, fmt.Errorf("ambiguous project %q matches %d projects", want, len(matches))

// latestSession returns the most recent interactive (non-agent) session,
// optionally scoped to a project. Agent-spawned transcripts are skipped so
// --latest targets the user's own sessions.
func latestSession(opts *options, root string) (discovery.SessionInfo, bool, error) {
	groups, err := discovery.Scan(root)
	if err != nil {
		return discovery.SessionInfo{}, false, err
	}
	if opts.project != "" {
		idx, err := discovery.FindProject(groups, opts.project)
		if err != nil {
			return discovery.SessionInfo{}, false, err
		}
		groups = groups[idx : idx+1]
	}
	for _, g := range groups {
		for _, s := range g.Sessions {
			if !s.IsAgent {
				return s, false, nil
			}
		}
	}
	return discovery.SessionInfo{}, false, fmt.Errorf("no sessions found")
}

// generate parses, redacts, and renders a session, returning the output dir.
func generate(opts *options, si discovery.SessionInfo) (string, error) {
	outDir := resolveOutDir(opts, si)
	if err := ensureOutDir(outDir, opts.force); err != nil {
		return "", err
	}
	sess, err := transcript.ParseFile(si.FilePath, transcript.Options{
		IncludeSubagents: opts.includeSubagents,
	})
	if err != nil {
		return "", err
	}
	if si.ID != "" {
		sess.ID = si.ID
	}
	if sess.SkippedLines > 0 {
		fmt.Fprintf(os.Stderr, "warning: skipped %d malformed line(s)\n", sess.SkippedLines)
	}
	if !opts.noRedact {
		home, _ := os.UserHomeDir()
		redact.Session(&sess, home)
	}
	if err := render.Site(sess, outDir, render.Options{Title: opts.title}); err != nil {
		return "", err
	}
	return outDir, nil
}

func ensureOutDir(dir string, force bool) error {
	entries, err := os.ReadDir(dir)
	if err != nil {
		if os.IsNotExist(err) {
			return os.MkdirAll(dir, 0o755)
		}
		return err
	}
	if len(entries) > 0 && !force {
		return fmt.Errorf("output directory %q is not empty (use --force to overwrite)", dir)
	}
	return nil
}

Important — latestSession does not return the globally newest interactive session

File: cmd/ccwhid/run.go:54-63 (the skip loop in latestSession)

Spec item 4 says --latest should "target the newest interactive session (in or out of a project scope)." The implementation instead returns the first non-agent session in group order, and groups are ordered by each group's newest session including agents (Scan sorts by groups[i].Sessions[0].ModTime, and Sessions[0] may be an agent transcript). So it finds "the newest interactive session inside whichever project had the most recent activity of any kind," not the newest interactive session overall.

Failure scenario (no --project):

  • Project A sessions: agent@T=100, interactive@T=50
  • Project B sessions: interactive@T=90

Scan orders A before B (100 > 90). The loop enters A, skips the agent, and returns interactive@T=50. But the newest interactive session is B's @T=90. --latest renders the wrong (older) session. This is very plausible in practice: running a code-review/Task agent in project A writes a fresh top-level transcript with a recent mtime, while your actual latest human session was in project B.

Note the old code (return groups[0].Sessions[0]) was correct for "newest session" because the group sort key is Sessions[0].ModTime; adding the agent-skip broke that invariant without switching to a global max.
