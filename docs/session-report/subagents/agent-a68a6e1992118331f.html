113	// Package discovery finds Claude Code session transcripts under
114	// ~/.claude/projects and builds a lightweight index for listing and selection.
115	package discovery
116	
117	import (
118		"bufio"
119		"encoding/json"
120		"fmt"
121		"os"
122		"path/filepath"
123		"sort"
124		"strings"
125		"time"
126	)
127	
128	// SessionInfo is a lightweight index entry for one transcript file. It is built
129	// by a cheap scan and does not parse the full timeline.
130	type SessionInfo struct {
131		ID           string
132		FilePath     string
133		ProjectDir   string // encoded directory name
134		ProjectPath  string // decoded absolute path
135		Title        string
136		FirstPrompt  string
137		ModTime      time.Time
138		MessageCount int
139		HasSubagents bool
140	}
141	
142	// ProjectGroup groups a project's sessions.
143	type ProjectGroup struct {
144		ProjectDir  string
145		ProjectPath string
146		Sessions    []SessionInfo
147	}
148	
149	// EncodeProjectDir maps an absolute cwd to its ~/.claude/projects dir name.
150	func EncodeProjectDir(cwd string) string {
151		return strings.ReplaceAll(cwd, "/", "-")
152	}
153	
154	// DecodeProjectDir maps a projects dir name back to an absolute path.
155	func DecodeProjectDir(name string) string {
156		return strings.ReplaceAll(name, "-", "/")
157	}
158	
159	// DefaultRoot returns ~/.claude/projects.
160	func DefaultRoot() (string, error) {
161		home, err := os.UserHomeDir()
162		if err != nil {
163			return "", err
164		}
165		return filepath.Join(home, ".claude", "projects"), nil
166	}
167	
168	// scanLine is the subset of fields needed to index a session cheaply.
169	type scanLine struct {
170		Type        string          `json:"type"`
171		AiTitle     string          `json:"aiTitle"`
172		IsSidechain bool            `json:"isSidechain"`
173		Timestamp   string          `json:"timestamp"`
174		Message     json.RawMessage `json:"message"`
175	}
176	
177	// indexFile reads a jsonl file and produces a SessionInfo without a full parse.
178	func indexFile(path, projectDir string) (SessionInfo, error) {
179		f, err := os.Open(path)
180		if err != nil {
181			return SessionInfo{}, err
182		}
183		defer f.Close()
184	
185		fi, err := f.Stat()
186		if err != nil {
187			return SessionInfo{}, err
188		}
189	
190		info := SessionInfo{
191			ID:          strings.TrimSuffix(filepath.Base(path), ".jsonl"),
192			FilePath:    path,
193			ProjectDir:  projectDir,
194			ProjectPath: DecodeProjectDir(projectDir),
195			ModTime:     fi.ModTime(),
196		}
197	
198		sc := bufio.NewScanner(f)
199		sc.Buffer(make([]byte, 0, 1024*1024), 16*1024*1024)
200		for sc.Scan() {
201			var l scanLine
202			if err := json.Unmarshal(sc.Bytes(), &l); err != nil {
203				continue // tolerate malformed lines
204			}
205			switch l.Type {
206			case "ai-title":
207				if l.AiTitle != "" {
208					info.Title = l.AiTitle
209				}
210			case "user", "assistant":
211				info.MessageCount++
212				if l.IsSidechain {
213					info.HasSubagents = true
214				}
215				if l.Type == "user" && info.FirstPrompt == "" {
216					info.FirstPrompt = firstPromptText(l.Message)
217				}
218			}
219		}
220		if info.Title == "" {
221			info.Title = truncate(info.FirstPrompt, 60)
222		}
223		return info, nil
224	}
225	
226	// firstPromptText extracts a plain-text preview from a user message whose
227	// content is either a JSON string or an array of blocks.
228	func firstPromptText(raw json.RawMessage) string {
229		if len(raw) == 0 {
230			return ""
231		}
232		var msg struct {
233			Content json.RawMessage `json:"content"`
234		}
235		if err := json.Unmarshal(raw, &msg); err != nil {
236			return ""
237		}
238		var s string
239		if json.Unmarshal(msg.Content, &s) == nil {
240			return strings.TrimSpace(s)
241		}
242		var blocks []struct {
243			Type string `json:"type"`
244			Text string `json:"text"`
245		}
246		if json.Unmarshal(msg.Content, &blocks) == nil {
247			for _, b := range blocks {
248				if b.Type == "text" {
249					return strings.TrimSpace(b.Text)
250				}
251			}
252		}
253		return ""
254	}
255	
256	func truncate(s string, n int) string {
257		s = strings.TrimSpace(s)
258		if len(s) <= n {
259			return s
260		}
261		return s[:n-1] + "…"
262	}
263	
264	// Scan indexes every session under root (the projects directory), grouped by
265	// project. Groups are ordered by most-recent session first; sessions within a
266	// group are newest-first.
267	func Scan(root string) ([]ProjectGroup, error) {
268		entries, err := os.ReadDir(root)
269		if err != nil {
270			return nil, err
271		}
272		var groups []ProjectGroup
273		for _, e := range entries {
274			if !e.IsDir() {
275				continue
276			}
277			dir := filepath.Join(root, e.Name())
278			files, err := filepath.Glob(filepath.Join(dir, "*.jsonl"))
279			if err != nil {
280				return nil, err
281			}
282			var sessions []SessionInfo
283			for _, fp := range files {
284				si, err := indexFile(fp, e.Name())
285				if err != nil {
286					continue
287				}
288				sessions = append(sessions, si)
289			}
290			if len(sessions) == 0 {
291				continue
292			}
293			sort.Slice(sessions, func(i, j int) bool {
294				return sessions[i].ModTime.After(sessions[j].ModTime)
295			})
296			groups = append(groups, ProjectGroup{
297				ProjectDir:  e.Name(),
298				ProjectPath: DecodeProjectDir(e.Name()),
299				Sessions:    sessions,
300			})
301		}
302		sort.Slice(groups, func(i, j int) bool {
303			return groups[i].Sessions[0].ModTime.After(groups[j].Sessions[0].ModTime)
304		})
305		return groups, nil
306	}
307	
308	// FindSession resolves a session id or unambiguous prefix across all projects.
309	func FindSession(root, idOrPrefix string) (SessionInfo, error) {
310		groups, err := Scan(root)
311		if err != nil {
312			return SessionInfo{}, err
313		}
314		var matches []SessionInfo
315		for _, g := range groups {
316			for _, s := range g.Sessions {
317				if s.ID == idOrPrefix {
318					return s, nil // exact wins immediately
319				}
320				if strings.HasPrefix(s.ID, idOrPrefix) {
321					matches = append(matches, s)
322				}
323			}
324		}
325		switch len(matches) {
326		case 0:
327			return SessionInfo{}, fmt.Errorf("no session matching %q", idOrPrefix)
328		case 1:
329			return matches[0], nil
330		default:
331			return SessionInfo{}, fmt.Errorf("ambiguous prefix %q matches %d sessions", idOrPrefix, len(matches))
332		}
333	}

// Package model defines the domain types shared across ccwhid: a parsed
// Session and the timeline of Turns, Blocks and ToolCalls within it.
package model

import "time"

type TurnKind string

const (
	TurnUser      TurnKind = "user"
	TurnAssistant TurnKind = "assistant"
)

type BlockType string

const (
	BlockText     BlockType = "text"
	BlockThinking BlockType = "thinking"
	BlockToolUse  BlockType = "tool_use"
)

// Session is one fully-parsed Claude Code transcript.
type Session struct {
	ID           string
	ProjectPath  string // decoded absolute cwd
	Title        string
	GitBranch    string
	Version      string
	StartedAt    time.Time
	EndedAt      time.Time
	Turns        []Turn
	SkippedLines int // count of malformed lines skipped while parsing
}

func (s Session) Duration() time.Duration { return s.EndedAt.Sub(s.StartedAt) }

func (s Session) DisplayTitle() string {
	if s.Title != "" {
		return s.Title
	}
	return "Untitled session"
}

// Turn is a single user or assistant message, holding ordered content blocks.
type Turn struct {
	Kind      TurnKind
	Timestamp time.Time
	Blocks    []Block
}

// Block is one content unit: assistant/user text, a thinking block, or a tool call.
type Block struct {
	Type BlockType
	Text string    // for BlockText and BlockThinking
	Tool *ToolCall // for BlockToolUse
}

// ToolCall is a single tool invocation and its result.
type ToolCall struct {
	ID        string
	Name      string // e.g. "Bash", "Edit", "Read", "Task"
	Summary   string // one-line summary for the collapsed card header
	InputJSON string // pretty-printed input for generic display
	Result    *ToolResult
	Diff      *Diff      // set for Edit/Write
	Subagents []Subagent // set for Task calls with sidechain activity
}

type ToolResult struct {
	Content string
	IsError bool
}

type Diff struct {
	Path    string
	OldText string
	NewText string
}

// Subagent is a nested sidechain run attached to a parent Task tool call.
type Subagent struct {
	Description string
	Turns       []Turn
}
