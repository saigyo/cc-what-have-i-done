700	+	Input        int
701	+	Output       int
702	+	CacheRead    int
703	+	CacheWrite5m int
704	+	CacheWrite1h int
705	+}
706	+
707	+// InOut is the base input+output total, excluding cache tokens. Used for the
708	+// report's headline figure so cache-read volume does not dominate it.
709	+func (t TokenCounts) InOut() int { return t.Input + t.Output }
710	+
711	+func (t *TokenCounts) add(u model.Usage) {
712	+	t.Input += u.Input
713	+	t.Output += u.Output
714	+	t.CacheRead += u.CacheRead
715	+	t.CacheWrite5m += u.CacheWrite5m
716	+	t.CacheWrite1h += u.CacheWrite1h
717	+}
718	+
719	+// ModelUsage is the token total and cost for one model in a session.
720	+type ModelUsage struct {
721	+	Model   string
722	+	Tokens  TokenCounts
723	+	CostUSD *float64 // nil when the model is unpriced
724	+}
725	+
726	+// Report is the computed usage for a session.
727	+type Report struct {
728	+	Total           TokenCounts
729	+	TotalCostUSD    *float64        // nil only when nothing was priced
730	+	ByModel         []ModelUsage    // sorted, highest in+out first
731	+	PerTurnCost     map[int]*float64 // top-level turn index -> cost (nil = unpriced)
732	+	PricesAsOf      string
733	+	HasUnknownModel bool // some tokens came from an unpriced model
734	+	HasAnyUsage     bool // at least one turn carried usage
735	+}

737	+// cost returns the USD cost of a token bucket at a model's price.
738	+func cost(t TokenCounts, p Price) float64 {
739	+	inRate := p.Input / 1e6
740	+	return float64(t.Input)*inRate +
741	+		float64(t.Output)*(p.Output/1e6) +
742	+		float64(t.CacheWrite5m)*inRate*cacheWrite5mMult +
743	+		float64(t.CacheWrite1h)*inRate*cacheWrite1hMult +
744	+		float64(t.CacheRead)*inRate*cacheReadMult
745	+}

747	+// Compute aggregates token usage and estimated cost for a session. Totals and
748	+// per-model figures include nested subagent turns; per-turn costs are recorded
749	+// for top-level turns only.
750	+func Compute(s model.Session) Report {
751	+	r := Report{PricesAsOf: PricesAsOf, PerTurnCost: map[int]*float64{}}
752	+	byModel := map[string]*TokenCounts{}
753	+	var order []string
754	+
755	+	accumulate := func(t model.Turn) {
756	+		if t.Usage == nil {
757	+			return
758	+		}
759	+		r.HasAnyUsage = true
760	+		r.Total.add(*t.Usage)
761	+		tc, ok := byModel[t.Model]
762	+		if !ok {
763	+			tc = &TokenCounts{}
764	+			byModel[t.Model] = tc
765	+			order = append(order, t.Model)
766	+		}
767	+		tc.add(*t.Usage)
768	+	}
769	+
770	+	// Totals + per-model: walk top-level and nested subagent turns.
771	+	for _, t := range s.Turns {
772	+		walk(t, accumulate)
773	+	}
774	+
775	+	// Per-turn cost for top-level turns only.
776	+	for i, t := range s.Turns {
777	+		if t.Usage == nil {
778	+			continue
779	+		}
780	+		if p, ok := Lookup(t.Model); ok {
781	+			c := cost(counts(*t.Usage), p)
782	+			r.PerTurnCost[i] = &c
783	+		} else {
784	+			r.PerTurnCost[i] = nil
785	+		}
786	+	}
787	+
788	+	// Per-model rows + grand total cost.
789	+	var total float64
790	+	priced := false
791	+	for _, m := range order {
792	+		tc := *byModel[m]
793	+		mu := ModelUsage{Model: m, Tokens: tc}
794	+		if p, ok := Lookup(m); ok {
795	+			c := cost(tc, p)
796	+			mu.CostUSD = &c
797	+			total += c
798	+			priced = true
799	+		} else {
800	+			r.HasUnknownModel = true
801	+		}
802	+		r.ByModel = append(r.ByModel, mu)
803	+	}
804	+	if priced {
805	+		r.TotalCostUSD = &total
806	+	}
807	+	sort.SliceStable(r.ByModel, func(i, j int) bool {
808	+		return r.ByModel[i].Tokens.InOut() > r.ByModel[j].Tokens.InOut()
809	+	})
810	+	return r
811	+}

1	# ccwhid Usage & Cost Section — Design
2	
3	**Status:** approved (2026-07-12)
4	
5	**Goal:** Add an opt-in report section that shows token usage and an estimated
6	cost for a rendered session, computed entirely from the `message.usage` data
7	already present in Claude Code transcripts — comparable to `/usage` or the
8	`ccusage` tool, but self-contained and offline.
9	
10	## Motivation
11	
12	Every assistant record in a transcript carries a full `usage` object and the
13	`model` id, e.g.:
14	
15	```json
16	"model": "claude-opus-4-8",
17	"usage": {
18	  "input_tokens": 10223,
19	  "output_tokens": 192,
20	  "cache_creation_input_tokens": 3008,
21	  "cache_read_input_tokens": 18878,
22	  "cache_creation": { "ephemeral_1h_input_tokens": 3008, "ephemeral_5m_input_tokens": 0 },
23	  "server_tool_use": { "web_search_requests": 0, "web_fetch_requests": 0 }
24	}
25	```
26	
27	Token counts are therefore **exact and free to compute**. Only cost requires a
28	price table, which is the one estimated element of the feature.
29	
30	## Scope
31	
32	- **Opt-in** via a `--usage` flag (default off). The section is only computed
33	  and rendered when requested.
34	- Displays, when enabled:
35	  1. A **collapsible "Usage" card** near the top of the report (summary +
36	     per-model breakdown).
37	  2. A **per-turn cost badge** on each assistant turn header.
38	- Cost is an **estimate** from an embedded, dated price table.
39	
40	### Out of scope for v1
41	
42	- **Server-tool request fees** (web search / web fetch requests have their own
43	  per-request pricing). v1 costs LLM token usage only; excluded fees are called
44	  out in the report footnote and revisited later (see Future Ideas).
45	- Live/remote price fetching at runtime (the tool is offline by design).
46	
47	## Data Model
48	
49	Extend the domain model so usage travels with each turn:
50	
51	- `model.Usage{ Input, Output, CacheRead, CacheWrite5m, CacheWrite1h int }`
52	- `model.Turn` gains `Model string` and `Usage *Usage`.
53	
54	The `cache_creation.ephemeral_5m_input_tokens` / `ephemeral_1h_input_tokens`
55	split is preserved because 5-minute and 1-hour cache writes are priced
56	differently. If a record only has the aggregate `cache_creation_input_tokens`
57	and no split, the whole amount is treated as a 5-minute write (the cheaper,
58	default TTL).
59	
60	## Transcript Parsing
61	
62	`internal/transcript` decodes `message.model` and `message.usage` for each
63	**assistant** record and attaches a `model.Usage` (plus `Model`) to the turn it
64	produces. Records without a `usage` object yield `Usage == nil` (no badge, not
65	counted). User records carry no usage.
66	
67	## Pricing (`internal/usage`)
68	
69	- An **embedded static table** (`prices.json`, `go:embed`) keyed by model id.
70	  Each entry gives per-million-token USD rates:
71	  `input`, `output`, `cacheWrite5m`, `cacheWrite1h`, `cacheRead`.
72	- A `PricesAsOf` date string (e.g. `"2026-07"`) is embedded and surfaced in the
73	  report footnote.
74	- The table is **populated from Anthropic's published list prices at
75	  implementation time** (a dev-time lookup to embed accurate numbers); the
76	  running binary performs no network access. Refreshing prices is a manual
77	  edit + release.
78	- Model resolution is exact-id first. `<synthetic>` and any id absent from the
79	  table resolve to **no price** (tokens still counted; cost shown as `n/a`).
80	
81	## Computation (`internal/usage`)
82	
83	`Compute(session) Report` walks the session's turns (main chain and, when
84	included, nested subagent turns) and aggregates:
85	
86	- **Per turn:** token totals and, if the model is priced, a cost.
87	- **Per model:** summed tokens and cost (or `nil` cost when unpriced).
88	- **Grand totals:** summed tokens; total cost = sum of priced models only.
89	
90	Cost for a model = Σ over token types of `tokens_type × rate_type ÷ 1_000_000`,
91	using the 5m/1h split for cache writes.
92	
93	```
94	Report{
95	  Total        TokenCounts
96	  TotalCostUSD *float64      // nil only if nothing was priced
97	  ByModel      []ModelUsage  // { Model, Tokens TokenCounts, CostUSD *float64 }
98	  PerTurnCost  map[int]*float64 // turn index → cost (nil = unpriced/no usage)
99	  PricesAsOf   string
100	  HasUnknownModel bool       // some tokens came from an unpriced model
101	  HasAnyUsage  bool          // false → transcript carried no usage at all
102	}
103	```
104	
105	## Rendering (`internal/render`)
106	
107	Rendering is gated on `render.Options{ Usage bool }`.
108	
109	**Usage card** (collapsible, placed after the report meta line):
110	
111	- Collapsed one-liner:
112	  `▸ Usage · 973k in+out · ~$12.40 (est.)`
113	  The headline token figure is **input + output only**, so it is not dominated
114	  by cache-read volume (which can be orders of magnitude larger).
115	- Expanded body:
116	  - Token breakdown: input / output / cache read / cache write.
117	  - Per-model table: `model · tokens · cost` (or `n/a`).
118	  - Muted footnote:
119	    *"Estimated — Anthropic list prices as of `<PricesAsOf>`; excludes
120	    server-tool fees."* When `HasUnknownModel`, append: *"totals exclude
121	    unpriced models (shown as n/a)."*
122	
123	**Per-turn badge:** a small muted element in each assistant turn header, e.g.
124	`· 12.3k tok · ~$0.18`. Unpriced/synthetic turns show `· 12.3k tok` (no cost).
125	Turns without usage show no badge.
126	
127	**No-data case:** with `--usage` set but `HasAnyUsage == false`, render a small
128	muted note ("no token-usage data in this transcript") instead of an empty card,
129	so the flag is never silently inert.
130	
131	Styling reuses existing report design tokens (warm palette, muted text, the
132	same collapse mechanism as tool cards); no new external assets, preserving the
133	self-contained/offline guarantee. Usage numbers are not secrets, so redaction
134	does not touch them.
135

137	
138	- **CLI:** add `--usage` (bool, default false) wired to `render.Options.Usage`.
139	- **TUI:** add an "Include usage & cost" toggle to the options screen (default

143

146	  badge; parsing never fails on absent fields.

148	  existing tolerance), not a fatal error.

314	// session from the per-turn usage data attached by the transcript parser.
315	package usage
316	
317	import (
318		_ "embed"
319		"encoding/json"
320	)
321	
322	//go:embed prices.json
323	var pricesJSON []byte
324	
325	// PricesAsOf is the month the embedded list prices were captured. Surfaced in
326	// the report so readers know how current the cost estimate is.
327	const PricesAsOf = "2026-07"
328	
329	// Cache pricing multipliers relative to a model's base input price. These are
330	// universal across Claude models (Anthropic prompt-caching pricing).
331	const (
332		cacheWrite5mMult = 1.25
333		cacheWrite1hMult = 2.0
334		cacheReadMult    = 0.1
335	)
336	
337	// Price holds a model's per-million-token USD rates for base input and output.
338	type Price struct {
339		Input  float64 `json:"input"`
340		Output float64 `json:"output"`
341	}
342	
343	var prices map[string]Price
344	
345	func init() {
346		if err := json.Unmarshal(pricesJSON, &prices); err != nil {
347			panic("usage: invalid embedded prices.json: " + err.Error())
348		}
349	}
350	
351	// Lookup returns the price for a model id and whether it is known. A trailing
352	// "-YYYYMMDD" date suffix (used by some model ids) is stripped on a miss.
353	func Lookup(modelID string) (Price, bool) {
354		if p, ok := prices[modelID]; ok {
355			return p, true
356		}
357		if norm := stripDateSuffix(modelID); norm != modelID {
358			if p, ok := prices[norm]; ok {
359				return p, true
360			}
361		}
362		return Price{}, false
363	}
364	
365	// stripDateSuffix removes a trailing "-YYYYMMDD" (dash + 8 digits) from id.
366	func stripDateSuffix(id string) string {
367		if len(id) < 9 {
368			return id
369		}
370		tail := id[len(id)-9:]
371		if tail[0] != '-' {
372			return id
373		}
374		for _, c := range tail[1:] {
375			if c < '0' || c > '9' {
376				return id
377			}
378		}
379		return id[:len(id)-9]
380	}

532		t.CacheWrite5m += u.CacheWrite5m
533		t.CacheWrite1h += u.CacheWrite1h
534	}
535	
536	// ModelUsage is the token total and cost for one model in a session.
537	type ModelUsage struct {
538		Model   string
539		Tokens  TokenCounts
540		CostUSD *float64 // nil when the model is unpriced
541	}
542	
543	// Report is the computed usage for a session.
544	type Report struct {
545		Total           TokenCounts
546		TotalCostUSD    *float64        // nil only when nothing was priced
547		ByModel         []ModelUsage    // sorted, highest in+out first
548		PerTurnCost     map[int]*float64 // top-level turn index -> cost (nil = unpriced)
549		PricesAsOf      string
550		HasUnknownModel bool // some tokens came from an unpriced model
551		HasAnyUsage     bool // at least one turn carried usage
552	}
553	
554	// cost returns the USD cost of a token bucket at a model's price.
555	func cost(t TokenCounts, p Price) float64 {
556		inRate := p.Input / 1e6
557		return float64(t.Input)*inRate +
558			float64(t.Output)*(p.Output/1e6) +
559			float64(t.CacheWrite5m)*inRate*cacheWrite5mMult +
560			float64(t.CacheWrite1h)*inRate*cacheWrite1hMult +
561			float64(t.CacheRead)*inRate*cacheReadMult
562	}
563	
564	// Compute aggregates token usage and estimated cost for a session. Totals and
565	// per-model figures include nested subagent turns; per-turn costs are recorded
566	// for top-level turns only.
567	func Compute(s model.Session) Report {
568		r := Report{PricesAsOf: PricesAsOf, PerTurnCost: map[int]*float64{}}
569		byModel := map[string]*TokenCounts{}
570		var order []string
571	
572		accumulate := func(t model.Turn) {
573			if t.Usage == nil {
574				return
575			}
576			r.HasAnyUsage = true
577			r.Total.add(*t.Usage)
578			tc, ok := byModel[t.Model]
579			if !ok {
580				tc = &TokenCounts{}
581				byModel[t.Model] = tc
582				order = append(order, t.Model)
583			}
584			tc.add(*t.Usage)
585		}
586	
587		// Totals + per-model: walk top-level and nested subagent turns.
588		for _, t := range s.Turns {
589			walk(t, accumulate)
590		}
591	
592		// Per-turn cost for top-level turns only.
593		for i, t := range s.Turns {
594			if t.Usage == nil {
595				continue
596			}
597			if p, ok := Lookup(t.Model); ok {
598				c := cost(counts(*t.Usage), p)
599				r.PerTurnCost[i] = &c
600			} else {
601				r.PerTurnCost[i] = nil
602			}
603		}
604	
605		// Per-model rows + grand total cost.
606		var total float64
607		priced := false
608		for _, m := range order {
609			tc := *byModel[m]
610			mu := ModelUsage{Model: m, Tokens: tc}
611			if p, ok := Lookup(m); ok {
612				c := cost(tc, p)
613				mu.CostUSD = &c
614				total += c
615				priced = true
616			} else {
617				r.HasUnknownModel = true
618			}
619			r.ByModel = append(r.ByModel, mu)
620		}
621		if priced {
622			r.TotalCostUSD = &total
623		}
624		sort.SliceStable(r.ByModel, func(i, j int) bool {
625			return r.ByModel[i].Tokens.InOut() > r.ByModel[j].Tokens.InOut()
626		})
627		return r
628	}
629	
630	// counts converts a model.Usage into a TokenCounts.
631	func counts(u model.Usage) TokenCounts {
632		return TokenCounts{Input: u.Input, Output: u.Output, CacheRead: u.CacheRead,
633			CacheWrite5m: u.CacheWrite5m, CacheWrite1h: u.CacheWrite1h}
634	}
635	
636	// walk visits a turn and any turns nested in its Task subagents.
637	func walk(t model.Turn, fn func(model.Turn)) {
638		fn(t)
639		for _, b := range t.Blocks {
640			if b.Tool == nil {
641				continue
642			}
643			for _, sub := range b.Tool.Subagents {
644				for _, st := range sub.Turns {
645					walk(st, fn)
646				}
647			}
648		}
649	}

53	// Parse reads a transcript stream and folds it into a Session timeline.
54	func Parse(r io.Reader, opts Options) (model.Session, error) {
55		var s model.Session
56		// toolIndex maps a tool_use id to the ToolCall pointer so results and
57		// sidechains can be attached after the fact.
58		toolIndex := map[string]*model.ToolCall{}
59		// lastTask tracks the most recent Task tool call; sidechainOwner is the Task
60		// whose Subagents any subsequent sidechain turns are appended to.
61		var lastTask *model.ToolCall
62		sidechainOwner := lastTask
63	
64		sc := bufio.NewScanner(r)
65		sc.Buffer(make([]byte, 0, 1024*1024), 64*1024*1024)
66		for sc.Scan() {
67			line := sc.Bytes()
68			if len(trimSpace(line)) == 0 {
69				continue
70			}
71			var rec rawRecord
72			if err := json.Unmarshal(line, &rec); err != nil {
73				s.SkippedLines++
74				continue
75			}
76			switch rec.Type {
77			case "ai-title":
78				if rec.AiTitle != "" {
79					s.Title = rec.AiTitle
80				}
81				continue
82			case "user", "assistant":
83				// fallthrough to handling below
84			default:
85				continue // ignore mode/permission/attachment/etc. for the timeline
86			}
87			if rec.IsMeta {
88				continue
89			}
90	
91			// Capture session-level metadata from the first record that has it.
92			if s.GitBranch == "" && rec.GitBranch != "" {
93				s.GitBranch = rec.GitBranch
94			}
95			if s.Version == "" && rec.Version != "" {
96				s.Version = rec.Version
97			}
98			if s.ProjectPath == "" && rec.Cwd != "" {
99				s.ProjectPath = rec.Cwd
100			}
101			ts := parseTime(rec.Timestamp)
102			if !ts.IsZero() {
103				if s.StartedAt.IsZero() {
104					s.StartedAt = ts
105				}
106				s.EndedAt = ts
107			}
108	
109			_, blocks := decodeMessageContent(rec.Message)
110	
111			// Sidechain (subagent) records.
112			if rec.IsSidechain {
113				if !opts.IncludeSubagents || sidechainOwner == nil {
114					continue
115				}
116				turn := buildTurn(rec, blocks, toolIndex)
117				if turn == nil {
118					continue
119				}
120				sub := &sidechainOwner.Subagents[len(sidechainOwner.Subagents)-1]
121				sub.Turns = append(sub.Turns, *turn)
122				continue
123			}
124	
125			// Main-chain records.
126			turn := buildTurn(rec, blocks, toolIndex)
127			if turn == nil {
128				continue // e.g. a user record that only carried a tool_result
129			}
130			s.Turns = append(s.Turns, *turn)
131	
132			// Seed a subagent slot for each Task tool call and make it the current
133			// sidechain owner. Tool-use ids are already registered in buildTurn.
134			// Note: sidechain records are attributed to the nearest preceding Task
135			// by file order. Sequential Task calls attribute correctly; multiple
136			// Task calls dispatched in parallel from one assistant message cannot be
137			// disambiguated from the raw records and will all attach to the last
138			// one — a known v1 limitation.
139			if opts.IncludeSubagents {
140				for i := range turn.Blocks {
141					tool := turn.Blocks[i].Tool
142					if tool != nil && tool.Name == "Task" {
143						tool.Subagents = append(tool.Subagents, model.Subagent{
144							Description: taskDescription(tool),
145						})
146						lastTask = tool
147						sidechainOwner = lastTask
148					}
149				}
150			}
151		}
152		if err := sc.Err(); err != nil {
153			// A single over-long line (bufio.ErrTooLong) is a pathological record,
154			// not a reason to discard the whole report: count it and return what we
155			// parsed so far.
156			if errors.Is(err, bufio.ErrTooLong) {
157				s.SkippedLines++
158				return s, nil
159			}
160			return s, err
161		}
162		return s, nil
163	}

186	// buildTurn converts a record's blocks into a model.Turn. tool_result blocks are
187	// attached to their originating ToolCall via toolIndex and do not themselves
188	// produce turn content. Returns nil if the record yields no displayable blocks.
189	func buildTurn(rec rawRecord, blocks []apiBlock, toolIndex map[string]*model.ToolCall) *model.Turn {
190		turn := &model.Turn{Timestamp: parseTime(rec.Timestamp)}
191		if rec.Type == "user" {
192			turn.Kind = model.TurnUser
193		} else {
194			turn.Kind = model.TurnAssistant
195		}
196		for _, b := range blocks {
197			switch b.Type {
198			case "text":
199				if b.Text != "" {
200					turn.Blocks = append(turn.Blocks, model.Block{Type: model.BlockText, Text: b.Text})
201				}
202			case "thinking":
203				if b.Thinking != "" {
204					turn.Blocks = append(turn.Blocks, model.Block{Type: model.BlockThinking, Text: b.Thinking})
205				}
206			case "tool_use":
207				tc := buildToolCall(b)
208				// Register the tool call so a later tool_result (in this turn's
209				// chain OR inside a sidechain) can be matched back to it by id.
210				toolIndex[tc.ID] = tc
211				turn.Blocks = append(turn.Blocks, model.Block{Type: model.BlockToolUse, Tool: tc})
212			case "tool_result":
213				if tc := toolIndex[b.ToolUseID]; tc != nil {
214					tc.Result = &model.ToolResult{Content: toolResultText(b.Content), IsError: b.IsError}
215				}
216			}
217		}
218		if len(turn.Blocks) == 0 {
219			return nil
220		}
221		if rec.Type == "assistant" {
222			turn.Model, turn.Usage = messageMeta(rec.Message)
223		}
224		return turn
225	}

227	func trimSpace(b []byte) []byte {
228		start, end := 0, len(b)
229		for start < end && (b[start] == ' ' || b[start] == '\t' || b[start] == '\r' || b[start] == '\n') {
230			start++
231		}
232		for end > start && (b[end-1] == ' ' || b[end-1] == '\t' || b[end-1] == '\r' || b[end-1] == '\n') {
233			end--
234		}
235		return b[start:end]
236	}

106		entries, err := os.ReadDir(dir)

109				return os.MkdirAll(dir, 0o755)

123				return err

125		}
