33	// stylesheet needed). lang may be empty for auto-detection.
34	func Highlight(code, lang string) template.HTML {
35		lexer := lexers.Get(lang)
36		if lexer == nil {
37			lexer = lexers.Analyse(code)
38		}
39		if lexer == nil {

45		}

48		if err != nil {
49			return template.HTML("<pre>" + html.EscapeString(code) + "</pre>")
50		}

52		if err := formatter.Format(&buf, style, it); err != nil {
