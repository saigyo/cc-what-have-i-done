101		cacheWrite1hMult = 2.0

104	
105	// Price holds a model's per-million-token USD rates for base input and output.
106	type Price struct {
107		Input  float64 `json:"input"`
108		Output float64 `json:"output"`
109	}
110	
111	var prices map[string]Price
112	
113	func init() {
114		if err := json.Unmarshal(pricesJSON, &prices); err != nil {
115			panic("usage: invalid embedded prices.json: " + err.Error())
116		}
117	}
118	
119	// Lookup returns the price for a model id and whether it is known. A trailing
120	// "-YYYYMMDD" date suffix (used by some model ids) is stripped on a miss.
121	func Lookup(modelID string) (Price, bool) {
122		if p, ok := prices[modelID]; ok {
123			return p, true
124		}
125		if norm := stripDateSuffix(modelID); norm != modelID {
126			if p, ok := prices[norm]; ok {
127				return p, true
128			}
129		}
130		return Price{}, false
131	}
132	
133	// stripDateSuffix removes a trailing "-YYYYMMDD" (dash + 8 digits) from id.
134	func stripDateSuffix(id string) string {
135		if len(id) < 9 {
136			return id
137		}
138		tail := id[len(id)-9:]
139		if tail[0] != '-' {
140			return id
141		}
142		for _, c := range tail[1:] {
143			if c < '0' || c > '9' {
144				return id
145			}
146		}
147		return id[:len(id)-9]
148	}
