98	// errNeedTUI signals that no session selector was provided and the caller
99	// should launch the interactive browser.
100	var errNeedTUI = errors.New("no session selected: launch TUI")
101	
102	func resolveOutDir(opts *options, si discovery.SessionInfo) string {
103		if opts.out != "" {
104			return opts.out
105		}
106		short := si.ID
107		if len(short) > 8 {
108			short = short[:8]
109		}
110		return filepath.Join("ccwhid-report", short)
111	}
112	
113	// resolveSession picks a session from flags. It returns needTUI=true when the
114	// user gave no selector.
115	func resolveSession(opts *options, root string) (discovery.SessionInfo, bool, error) {
116		switch {
117		case opts.session != "":
118			si, err := discovery.FindSession(root, opts.session)
119			return si, false, err
120		case opts.latest:
121			return latestSession(opts, root)
122		default:
123			return discovery.SessionInfo{}, true, nil
124		}
125	}
126	
127	func latestSession(opts *options, root string) (discovery.SessionInfo, bool, error) {
128		groups, err := discovery.Scan(root)
129		if err != nil {
130			return discovery.SessionInfo{}, false, err
131		}
132		for _, g := range groups {
133			if opts.project != "" && !matchProject(g, opts.project) {
134				continue
135			}
136			if len(g.Sessions) > 0 {
137				return g.Sessions[0], false, nil
138			}
139		}
140		return discovery.SessionInfo{}, false, fmt.Errorf("no sessions found")
141	}
142	
143	func matchProject(g discovery.ProjectGroup, want string) bool {
144		return g.ProjectPath == want ||
145			strings.EqualFold(filepath.Base(g.ProjectPath), want) ||
146			strings.Contains(g.ProjectPath, want)
147	}
148	
149	// generate parses, redacts, and renders a session, returning the output dir.
150	func generate(opts *options, si discovery.SessionInfo) (string, error) {
151		outDir := resolveOutDir(opts, si)
152		if err := ensureOutDir(outDir, opts.force); err != nil {
153			return "", err
154		}
155		sess, err := transcript.ParseFile(si.FilePath, transcript.Options{
156			IncludeSubagents: opts.includeSubagents,
157		})
158		if err != nil {
159			return "", err
160		}
161		if si.ID != "" {
162			sess.ID = si.ID
163		}
164		if !opts.noRedact {
165			home, _ := os.UserHomeDir()
166			redact.Session(&sess, home)
167		}
168		if err := render.Site(sess, outDir, render.Options{Title: opts.title}); err != nil {
169			return "", err
170		}
171		return outDir, nil
172	}
173	
174	func ensureOutDir(dir string, force bool) error {
175		entries, err := os.ReadDir(dir)
176		if err != nil {
177			if os.IsNotExist(err) {
178				return os.MkdirAll(dir, 0o755)
179			}
180			return err
181		}
182		if len(entries) > 0 && !force {
183			return fmt.Errorf("output directory %q is not empty (use --force to overwrite)", dir)
184		}
185		return nil
186	}
187	
188	// openInBrowser opens path (a file or dir) in the default browser.
189	func openInBrowser(path string) error {
190		index := filepath.Join(path, "index.html")
191		abs, err := filepath.Abs(index)
192		if err != nil {
193			return err
194		}
195		url := "file://" + abs
196		switch runtime.GOOS {
197		case "darwin":
198			return exec.Command("open", url).Start()
199		case "windows":
200			return exec.Command("rundll32", "url.dll,FileProtocolHandler", url).Start()
201		default:
202			return exec.Command("xdg-open", url).Start()
203		}
204	}
