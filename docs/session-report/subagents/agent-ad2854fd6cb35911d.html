166	func sessionIDFromPath(path string) string {
167		base := path
168		if i := lastSlash(path); i >= 0 {
169			base = path[i+1:]
170		}
171		if len(base) > len(".jsonl") && base[len(base)-len(".jsonl"):] == ".jsonl" {
172			return base[:len(base)-len(".jsonl")]
173		}
174		return base
175	}

177	func lastSlash(s string) int {
178		for i := len(s) - 1; i >= 0; i-- {
179			if s[i] == '/' {
180				return i
181			}
182		}
183		return -1
184	}

186	// Parse reads a transcript stream and folds it into a Session timeline.
187	func Parse(r io.Reader, opts Options) (model.Session, error) {
188		var s model.Session
189		// toolIndex maps a tool_use id to the ToolCall pointer so results and
190		// sidechains can be attached after the fact.
191		toolIndex := map[string]*model.ToolCall{}
192		// lastTask tracks the most recent Task tool call for sidechain attachment.
193		var lastTask *model.ToolCall
194		// subByFirstUUID lets sidechain turns append to the right subagent; keyed
195		// by the Task tool that owns them.
196		sidechainOwner := lastTask
197	
198		sc := bufio.NewScanner(r)
199		sc.Buffer(make([]byte, 0, 1024*1024), 64*1024*1024)
200		for sc.Scan() {
201			line := sc.Bytes()
202			if len(trimSpace(line)) == 0 {
203				continue
204			}
205			var rec rawRecord
206			if err := json.Unmarshal(line, &rec); err != nil {
207				s.SkippedLines++
208				continue
209			}
210			switch rec.Type {
211			case "ai-title":
212				if rec.AiTitle != "" {
213					s.Title = rec.AiTitle
214				}
215				continue
216			case "user", "assistant":
217				// fallthrough to handling below
218			default:
219				continue // ignore mode/permission/attachment/etc. for the timeline
220			}
221			if rec.IsMeta {
222				continue
223			}
224	
225			// Capture session-level metadata from the first record that has it.
226			if s.GitBranch == "" && rec.GitBranch != "" {
227				s.GitBranch = rec.GitBranch
228			}
229			if s.Version == "" && rec.Version != "" {
230				s.Version = rec.Version
231			}
232			if s.ProjectPath == "" && rec.Cwd != "" {
233				s.ProjectPath = rec.Cwd
234			}
235			ts := parseTime(rec.Timestamp)
236			if !ts.IsZero() {
237				if s.StartedAt.IsZero() {
238					s.StartedAt = ts
239				}
240				s.EndedAt = ts
241			}
242	
243			_, blocks := decodeMessageContent(rec.Message)
244	
245			// Sidechain (subagent) records.
246			if rec.IsSidechain {
247				if !opts.IncludeSubagents || sidechainOwner == nil {
248					continue
249				}
250				turn := buildTurn(rec, blocks, toolIndex)
251				if turn == nil {
252					continue
253				}
254				sub := &sidechainOwner.Subagents[len(sidechainOwner.Subagents)-1]
255				sub.Turns = append(sub.Turns, *turn)
256				continue
257			}
258	
259			// Main-chain records.
260			turn := buildTurn(rec, blocks, toolIndex)
261			if turn == nil {
262				continue // e.g. a user record that only carried a tool_result
263			}
264			s.Turns = append(s.Turns, *turn)
265	
266			// After appending, register any Task tool call as the new sidechain
267			// owner and seed its first subagent slot.
268			for i := range turn.Blocks {
269				b := &s.Turns[len(s.Turns)-1].Blocks[i]
270				if b.Tool != nil {
271					toolIndex[b.Tool.ID] = b.Tool
272					if b.Tool.Name == "Task" && opts.IncludeSubagents {
273						b.Tool.Subagents = append(b.Tool.Subagents, model.Subagent{
274							Description: taskDescription(b.Tool),
275						})
276						lastTask = b.Tool
277						sidechainOwner = lastTask
278					}
279				}
280			}
281		}
282		if err := sc.Err(); err != nil {
283			return s, err
284		}
285		return s, nil
286	}

288	// buildTurn converts a record's blocks into a model.Turn. tool_result blocks are
289	// attached to their originating ToolCall via toolIndex and do not themselves
290	// produce turn content. Returns nil if the record yields no displayable blocks.
291	func buildTurn(rec rawRecord, blocks []apiBlock, toolIndex map[string]*model.ToolCall) *model.Turn {
292		turn := &model.Turn{Timestamp: parseTime(rec.Timestamp)}
293		if rec.Type == "user" {
294			turn.Kind = model.TurnUser
295		} else {
296			turn.Kind = model.TurnAssistant
297		}
298		for _, b := range blocks {
299			switch b.Type {
300			case "text":
301				if b.Text != "" {
302					turn.Blocks = append(turn.Blocks, model.Block{Type: model.BlockText, Text: b.Text})
303				}
304			case "thinking":
305				if b.Thinking != "" {
306					turn.Blocks = append(turn.Blocks, model.Block{Type: model.BlockThinking, Text: b.Thinking})
307				}
308			case "tool_use":
309				turn.Blocks = append(turn.Blocks, model.Block{Type: model.BlockToolUse, Tool: buildToolCall(b)})
310			case "tool_result":
311				if tc := toolIndex[b.ToolUseID]; tc != nil {
312					tc.Result = &model.ToolResult{Content: toolResultText(b.Content), IsError: b.IsError}
313				}
314			}
315		}
316		if len(turn.Blocks) == 0 {
317			return nil
318		}
319		return turn
320	}

322	func trimSpace(b []byte) []byte {
323		start, end := 0, len(b)
324		for start < end && (b[start] == ' ' || b[start] == '\t' || b[start] == '\r' || b[start] == '\n') {
325			start++
326		}
327		for end > start && (b[end-1] == ' ' || b[end-1] == '\t' || b[end-1] == '\r' || b[end-1] == '\n') {
328			end--
329		}
330		return b[start:end]

341		tc := &model.ToolCall{
342			ID:        b.ID,

344			InputJSON: prettyJSON(b.Input),

347		tc.Summary = toolSummary(b.Name, input)
348		switch b.Name {

350			tc.Diff = buildDiff(b.Name, input)

353	}

355	func decodeInput(raw json.RawMessage) map[string]any {
356		m := map[string]any{}
357		if len(raw) > 0 {
358			_ = json.Unmarshal(raw, &m)
359		}
360		return m
361	}

364		if v, ok := m[key].(string); ok {
365			return v
366		}
367		return ""
368	}

370	// toolSummary produces a compact one-line header for a tool card.
371	func toolSummary(name string, in map[string]any) string {
372		switch name {
373		case "Bash":
374			if c := str(in, "command"); c != "" {
375				return firstLine(c)
376			}
377		case "Read", "Edit", "Write", "NotebookEdit":
378			if p := str(in, "file_path"); p != "" {
379				return p
380			}
381		case "Glob":
382			return str(in, "pattern")
383		case "Grep":
384			return str(in, "pattern")
385		case "Task":
386			return str(in, "description")
387		case "WebFetch":
388			return str(in, "url")
389		case "WebSearch":
390			return str(in, "query")
391		}
392		return ""
393	}

395	func buildDiff(name string, in map[string]any) *model.Diff {
396		d := &model.Diff{Path: str(in, "file_path")}
397		if name == "Write" {
398			d.NewText = str(in, "content")
399			return d
400		}
401		d.OldText = str(in, "old_string")
402		d.NewText = str(in, "new_string")
403		return d
404	}

414		for i := 0; i < len(s); i++ {
415			if s[i] == '\n' {

423		if len(raw) == 0 {
424			return ""

Important — internal/transcript/parse.go:134-146: sidechain misattribution when two Task calls precede sidechain output (named risk 3). sidechainOwner is repointed to whichever Task tool_use is processed last within the registration loop. If an assistant turn dispatches two parallel Task calls in one message (a supported Claude Code pattern for parallel subagents), or if a second Task is issued before the first one's tool_result/sidechain output has fully arrived, all interleaved sidechain records — including ones that actually belong to the first Task — get appended to the second Task's Subagents[0]. This is an inherent limitation of "attach to nearest preceding Task by file order" without any id-level correlation between a sidechain conversation and its spawning tool_use.id (the raw record format captured by Task 4's rawRecord has no such field). Worth flagging explicitly since it's a real ordering assumption baked into the algorithm, not just a theoretical edge case — genuinely realistic with concurrent subagent dispatch.

Minor — internal/transcript/parse.go:134-136: unnecessary re-indirection. The registration loop re-fetches b := &s.Turns[len(s.Turns)-1].Blocks[i] instead of just using &turn.Blocks[i] (or ranging over turn.Blocks directly) — turn.Blocks's backing array is the same one just copied into s.Turns via *turn, so this adds a layer of indirection with no functional difference.
