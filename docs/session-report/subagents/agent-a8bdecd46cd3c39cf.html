87	// text before it is rendered. It is best-effort defense-in-depth, not a
88	// guarantee of complete secret removal.
89	package redact
90	
91	import (
92		"regexp"
93		"strings"
94	
95		"github.com/saigyo/cc-what-have-i-done/internal/model"
96	)
97	
98	type rule struct {
99		kind string
100		re   *regexp.Regexp
101	}
102	
103	// Ordered: more specific patterns first.
104	var rules = []rule{
105		{"aws-key", regexp.MustCompile(`\bAKIA[0-9A-Z]{16}\b`)},
106		{"token", regexp.MustCompile(`\b(?:sk-[A-Za-z0-9]{20,}|ghp_[A-Za-z0-9]{36}|github_pat_[A-Za-z0-9_]{50,}|xox[baprs]-[A-Za-z0-9-]{10,})\b`)},
107		{"jwt", regexp.MustCompile(`\beyJ[A-Za-z0-9_-]{10,}\.[A-Za-z0-9_-]{10,}\.[A-Za-z0-9_-]{10,}\b`)},
108		{"assignment", regexp.MustCompile(`(?i)\b([A-Z0-9_]*(?:KEY|TOKEN|SECRET|PASSWORD|PASSWD))\s*[=:]\s*['"]?([^\s'"]{6,})`)},
109	}
110	
111	// Redactor applies redaction rules and a home-directory rewrite.
112	type Redactor struct {
113		homeDir string
114	}
115	
116	func New(homeDir string) *Redactor { return &Redactor{homeDir: homeDir} }
117	
118	// String redacts a single string.
119	func (r *Redactor) String(s string) string {
120		for _, ru := range rules {
121			if ru.kind == "assignment" {
122				s = ru.re.ReplaceAllString(s, "$1=[REDACTED:assignment]")
123				continue
124			}
125			s = ru.re.ReplaceAllString(s, "[REDACTED:"+ru.kind+"]")
126		}
127		if r.homeDir != "" {
128			s = strings.ReplaceAll(s, r.homeDir, "~")
129		}
130		return s
131	}
132	
133	// Session redacts every user-visible text field of a Session in place.
134	func Session(s *model.Session, homeDir string) {
135		r := New(homeDir)
136		for i := range s.Turns {
137			redactTurn(r, &s.Turns[i])
138		}
139	}
140	
141	func redactTurn(r *Redactor, t *model.Turn) {
142		for i := range t.Blocks {
143			b := &t.Blocks[i]
144			b.Text = r.String(b.Text)
145			if b.Tool != nil {
146				redactTool(r, b.Tool)
147			}
148		}
149	}
150	
151	func redactTool(r *Redactor, tc *model.ToolCall) {
152		tc.Summary = r.String(tc.Summary)
153		tc.InputJSON = r.String(tc.InputJSON)
154		if tc.Result != nil {
155			tc.Result.Content = r.String(tc.Result.Content)
156		}
157		if tc.Diff != nil {
158			tc.Diff.OldText = r.String(tc.Diff.OldText)
159			tc.Diff.NewText = r.String(tc.Diff.NewText)
160			tc.Diff.Path = r.String(tc.Diff.Path)
161		}
162		for i := range tc.Subagents {
163			for j := range tc.Subagents[i].Turns {
164				redactTurn(r, &tc.Subagents[i].Turns[j])
165			}
166		}
167	}
