// Package redact scrubs likely secrets and the user's home path from transcript
// text before it is rendered. It is best-effort defense-in-depth, not a
// guarantee of complete secret removal.
package redact

import (
	"regexp"
	"strings"

	"github.com/saigyo/cc-what-have-i-done/internal/model"
)

type rule struct {
	kind string
	re   *regexp.Regexp
}

// Ordered: more specific patterns first.
var rules = []rule{
	{"aws-key", regexp.MustCompile(`\bAKIA[0-9A-Z]{16}\b`)},
	{"token", regexp.MustCompile(`\b(?:sk-[A-Za-z0-9]{20,}|ghp_[A-Za-z0-9]{36}|github_pat_[A-Za-z0-9_]{50,}|xox[baprs]-[A-Za-z0-9-]{10,})\b`)},
	{"jwt", regexp.MustCompile(`\beyJ[A-Za-z0-9_-]{10,}\.[A-Za-z0-9_-]{10,}\.[A-Za-z0-9_-]{10,}\b`)},
	{"assignment", regexp.MustCompile(`(?i)\b([A-Z0-9_]*(?:KEY|TOKEN|SECRET|PASSWORD|PASSWD))\s*[=:]\s*['"]?([^\s'"]{6,})`)},
}

// Redactor applies redaction rules and a home-directory rewrite.
type Redactor struct {
	homeDir string
}

func New(homeDir string) *Redactor { return &Redactor{homeDir: homeDir} }

// String redacts a single string.
func (r *Redactor) String(s string) string {
	for _, ru := range rules {
		if ru.kind == "assignment" {
			s = ru.re.ReplaceAllString(s, "$1=[REDACTED:assignment]")
			continue
		}
		s = ru.re.ReplaceAllString(s, "[REDACTED:"+ru.kind+"]")
	}
	if r.homeDir != "" {
		s = strings.ReplaceAll(s, r.homeDir, "~")
	}
	return s
}

// Session redacts every user-visible text field of a Session in place.
func Session(s *model.Session, homeDir string) {
	r := New(homeDir)
	for i := range s.Turns {
		redactTurn(r, &s.Turns[i])
	}
}

func redactTurn(r *Redactor, t *model.Turn) {
	for i := range t.Blocks {
		b := &t.Blocks[i]
		b.Text = r.String(b.Text)
		if b.Tool != nil {
			redactTool(r, b.Tool)
		}
	}
}

func redactTool(r *Redactor, tc *model.ToolCall) {
	tc.Summary = r.String(tc.Summary)
	tc.InputJSON = r.String(tc.InputJSON)
	if tc.Result != nil {
		tc.Result.Content = r.String(tc.Result.Content)
	}
	if tc.Diff != nil {
		tc.Diff.OldText = r.String(tc.Diff.OldText)
		tc.Diff.NewText = r.String(tc.Diff.NewText)
		tc.Diff.Path = r.String(tc.Diff.Path)
	}
	for i := range tc.Subagents {
		for j := range tc.Subagents[i].Turns {
			redactTurn(r, &tc.Subagents[i].Turns[j])
		}
	}
}

53	func Parse(r io.Reader, opts Options) (model.Session, error) {

56		// sidechains can be attached after the fact.

58		// lastTask tracks the most recent Task tool call for sidechain attachment.
59		var lastTask *model.ToolCall

61		// by the Task tool that owns them.
62		sidechainOwner := lastTask

70			}

83				// fallthrough to handling below
84			default:

86			}

88				continue
89			}

92			if s.GitBranch == "" && rec.GitBranch != "" {
93				s.GitBranch = rec.GitBranch
94			}
95			if s.Version == "" && rec.Version != "" {
96				s.Version = rec.Version
97			}
98			if s.ProjectPath == "" && rec.Cwd != "" {
99				s.ProjectPath = rec.Cwd
100			}
101			ts := parseTime(rec.Timestamp)
102			if !ts.IsZero() {
103				if s.StartedAt.IsZero() {
104					s.StartedAt = ts
105				}
106				s.EndedAt = ts
107			}
108	
109			_, blocks := decodeMessageContent(rec.Message)
110	
111			// Sidechain (subagent) records.
112			if rec.IsSidechain {
113				if !opts.IncludeSubagents || sidechainOwner == nil {
114					continue
115				}
116				turn := buildTurn(rec, blocks, toolIndex)
117				if turn == nil {
118					continue
119				}
120				sub := &sidechainOwner.Subagents[len(sidechainOwner.Subagents)-1]
121				sub.Turns = append(sub.Turns, *turn)
122				continue
123			}
124	
125			// Main-chain records.
126			turn := buildTurn(rec, blocks, toolIndex)
127			if turn == nil {
128				continue // e.g. a user record that only carried a tool_result
129			}
130			s.Turns = append(s.Turns, *turn)
131	
132			// Seed a subagent slot for each Task tool call and make it the current
133			// sidechain owner. Tool-use ids are already registered in buildTurn.
134			// Note: sidechain records are attributed to the nearest preceding Task
135			// by file order. Sequential Task calls attribute correctly; multiple
136			// Task calls dispatched in parallel from one assistant message cannot be
137			// disambiguated from the raw records and will all attach to the last
138			// one — a known v1 limitation.
139			if opts.IncludeSubagents {
140				for i := range turn.Blocks {
141					tool := turn.Blocks[i].Tool

143						tool.Subagents = append(tool.Subagents, model.Subagent{
144							Description: taskDescription(tool),
145						})

148					}
149				}
150			}
151		}
152		if err := sc.Err(); err != nil {
153			return s, err
154		}

156	}

158	// buildTurn converts a record's blocks into a model.Turn. tool_result blocks are
159	// attached to their originating ToolCall via toolIndex and do not themselves
160	// produce turn content. Returns nil if the record yields no displayable blocks.
161	func buildTurn(rec rawRecord, blocks []apiBlock, toolIndex map[string]*model.ToolCall) *model.Turn {
162		turn := &model.Turn{Timestamp: parseTime(rec.Timestamp)}
163		if rec.Type == "user" {
164			turn.Kind = model.TurnUser
165		} else {
166			turn.Kind = model.TurnAssistant
167		}
168		for _, b := range blocks {
169			switch b.Type {
170			case "text":
171				if b.Text != "" {
172					turn.Blocks = append(turn.Blocks, model.Block{Type: model.BlockText, Text: b.Text})
173				}
174			case "thinking":
175				if b.Thinking != "" {
176					turn.Blocks = append(turn.Blocks, model.Block{Type: model.BlockThinking, Text: b.Thinking})
177				}
178			case "tool_use":

180				// Register the tool call so a later tool_result (in this turn's
181				// chain OR inside a sidechain) can be matched back to it by id.

186					tc.Result = &model.ToolResult{Content: toolResultText(b.Content), IsError: b.IsError}
187				}
188			}
189		}
190		if len(turn.Blocks) == 0 {
191			return nil
192		}
193		return turn
194	}

197		start, end := 0, len(b)

1	# ccwhid — cc-what-have-i-done
2	
3	Turn a Claude Code session transcript into a browsable, self-contained static
4	HTML report. See what you prompted and what Claude Code did — and commit the
5	report into your repo as a documentation artifact.
6	
7	> **Status:** under active development. See
8	> [`docs/superpowers/plans/`](docs/superpowers/plans/) for the implementation plan.
9	
10	## Install
11	
12	```bash
13	go install github.com/saigyo/cc-what-have-i-done/cmd/ccwhid@latest
14	```
15	
16	Or build locally:
17	
18	```bash
19	go build -o ccwhid ./cmd/ccwhid
20	```
21	
22	## Usage
23	
24	```bash
25	ccwhid                      # browse sessions in an interactive TUI
26	ccwhid --latest             # render the most recent session
27	ccwhid --session <id>       # render a specific session (id or prefix)
28	ccwhid --session <id> --open
29	```
30	
31	The report is written to `./ccwhid-report/<session-short>/` by default (override
32	with `--out`). Open `index.html` in any browser — no server needed.
33	
34	### Flags
35	
36	| Flag | Description |
37	|------|-------------|
38	| `--session <id>` | Session id or unambiguous prefix to render |
39	| `--project <path\|name>` | Scope `--latest` to a project |
40	| `--latest` | Render the most recent session |
41	| `--out <dir>` | Output directory |
42	| `--title <str>` | Override the report title |
43	| `--include-subagents` | Include subagent (Task) activity (default true; `--include-subagents=false` to omit) |
44	| `--no-redact` | Disable secret redaction |
45	| `--force` | Overwrite a non-empty output directory |
46	| `--open` | Open the report in a browser when done |
47	
48	## Redaction
49	
50	By default, ccwhid scrubs common secret shapes (AWS keys, API tokens, JWTs,
51	`KEY=[REDACTED:assignment] assignments) and rewrites your home directory to `~`.
52	This is best-effort defense-in-depth — **review generated reports before
53	committing them.** Disable with `--no-redact`.
54	
55	## What's included
56	
57	Your prompts, Claude's replies (rendered markdown + syntax-highlighted code),
58	tool calls with collapsible detail, `Edit`/`Write` diffs, collapsed thinking
59	blocks, and — optionally — nested subagent activity. System reminders and
60	attachments are omitted for readability.
61
