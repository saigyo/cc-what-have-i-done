111	import (
112		"bytes"

117		chromahtml "github.com/alecthomas/chroma/v2/formatters/html"
118		"github.com/alecthomas/chroma/v2/lexers"
119		"github.com/alecthomas/chroma/v2/styles"
120		"github.com/yuin/goldmark"
121		"github.com/yuin/goldmark/extension"
122	)
123	
124	var md = goldmark.New(goldmark.WithExtensions(extension.GFM))
125	
126	// Markdown renders CommonMark+GFM to sanitized-enough HTML. Input is trusted
127	// transcript content (the user's own session), so we do not strip HTML, but we
128	// do escape code via goldmark's default renderer.
129	func Markdown(src string) template.HTML {
130		var buf bytes.Buffer
131		if err := md.Convert([]byte(src), &buf); err != nil {
132			return template.HTML("<pre>" + html.EscapeString(src) + "</pre>")
133		}
134		return template.HTML(buf.String())
135	}
136	
137	// Highlight renders a code string with chroma using inline styles (no external
138	// stylesheet needed). lang may be empty for auto-detection.
139	func Highlight(code, lang string) template.HTML {
140		lexer := lexers.Get(lang)
141		if lexer == nil {
142			lexer = lexers.Analyse(code)
143		}
144		if lexer == nil {
145			lexer = lexers.Fallback
146		}
147		style := styles.Get("github")
148		if style == nil {
149			style = styles.Fallback
150		}
151		formatter := chromahtml.New(chromahtml.WithClasses(false), chromahtml.TabWidth(2))
152		it, err := lexer.Tokenise(nil, code)
153		if err != nil {
154			return template.HTML("<pre>" + html.EscapeString(code) + "</pre>")
155		}
156		var buf bytes.Buffer
157		if err := formatter.Format(&buf, style, it); err != nil {
158			return template.HTML("<pre>" + html.EscapeString(code) + "</pre>")
159		}
160		return template.HTML(buf.String())
161	}
162	
163	// DiffHTML renders a simple line-based diff of a Diff's old/new text.
164	func DiffHTML(d *model.Diff) template.HTML {
165		var b strings.Builder
166		b.WriteString(`<div class="diff">`)
167		if d.OldText != "" {
168			for _, line := range strings.Split(d.OldText, "\n") {
169				b.WriteString(`<div class="diff-del">- `)
170				b.WriteString(html.EscapeString(line))
171				b.WriteString("</div>")
172			}
173		}
174		if d.NewText != "" {
175			for _, line := range strings.Split(d.NewText, "\n") {
176				b.WriteString(`<div class="diff-add">+ `)
177				b.WriteString(html.EscapeString(line))
178				b.WriteString("</div>")
179			}
180		}
181		b.WriteString("</div>")
182		return template.HTML(b.String())
183	}
