570	+// indexFile reads a jsonl file and produces a SessionInfo without a full parse.
571	+func indexFile(path, projectDir string) (SessionInfo, error) {
572	+	f, err := os.Open(path)
573	+	if err != nil {
574	+		return SessionInfo{}, err
575	+	}
576	+	defer f.Close()
577	+
578	+	fi, err := f.Stat()
579	+	if err != nil {
580	+		return SessionInfo{}, err
581	+	}
582	+
583	+	info := SessionInfo{
584	+		ID:          strings.TrimSuffix(filepath.Base(path), ".jsonl"),
585	+		FilePath:    path,
586	+		ProjectDir:  projectDir,
587	+		ProjectPath: DecodeProjectDir(projectDir),
588	+		ModTime:     fi.ModTime(),
589	+	}
590	+
591	+	sc := bufio.NewScanner(f)
592	+	sc.Buffer(make([]byte, 0, 1024*1024), 16*1024*1024)
593	+	cwdSet := false
594	+	for sc.Scan() {
595	+		var l scanLine
596	+		if err := json.Unmarshal(sc.Bytes(), &l); err != nil {
597	+			continue // tolerate malformed lines
598	+		}
599	+		switch l.Type {
600	+		case "ai-title":
601	+			if l.AiTitle != "" {
602	+				info.Title = l.AiTitle
603	+			}
604	+		case "user", "assistant":
605	+			info.MessageCount++
606	+			if l.IsSidechain {
607	+				info.HasSubagents = true
608	+			}
609	+			if l.Type == "user" && info.FirstPrompt == "" {
610	+				info.FirstPrompt = firstPromptText(l.Message)
611	+			}
612	+			if !cwdSet && l.Cwd != "" {
613	+				info.ProjectPath = l.Cwd
614	+				cwdSet = true
615	+			}
616	+		}
617	+	}
618	+	if info.Title == "" {
619	+		info.Title = truncate(info.FirstPrompt, 60)
620	+	}
621	+	return info, nil
622	+}

624	+// firstPromptText extracts a plain-text preview from a user message whose
625	+// content is either a JSON string or an array of blocks.
626	+func firstPromptText(raw json.RawMessage) string {
627	+	if len(raw) == 0 {
628	+		return ""
629	+	}
630	+	var msg struct {
631	+		Content json.RawMessage `json:"content"`
632	+	}
633	+	if err := json.Unmarshal(raw, &msg); err != nil {
634	+		return ""
635	+	}
636	+	var s string
637	+	if json.Unmarshal(msg.Content, &s) == nil {
638	+		return strings.TrimSpace(s)
639	+	}
640	+	var blocks []struct {
641	+		Type string `json:"type"`
642	+		Text string `json:"text"`
643	+	}
644	+	if json.Unmarshal(msg.Content, &blocks) == nil {
645	+		for _, b := range blocks {
646	+			if b.Type == "text" {
647	+				return strings.TrimSpace(b.Text)
648	+			}
649	+		}
650	+	}
651	+	return ""

655	+	s = strings.TrimSpace(s)
656	+	r := []rune(s)

663	+// Scan indexes every session under root (the projects directory), grouped by
664	+// project. Groups are ordered by most-recent session first; sessions within a
665	+// group are newest-first.
666	+func Scan(root string) ([]ProjectGroup, error) {
667	+	entries, err := os.ReadDir(root)
668	+	if err != nil {
669	+		return nil, err
670	+	}
671	+	var groups []ProjectGroup
672	+	for _, e := range entries {
673	+		if !e.IsDir() {
674	+			continue
675	+		}
676	+		dir := filepath.Join(root, e.Name())
677	+		files, err := filepath.Glob(filepath.Join(dir, "*.jsonl"))
678	+		if err != nil {
679	+			return nil, err
680	+		}
681	+		var sessions []SessionInfo
682	+		for _, fp := range files {
683	+			si, err := indexFile(fp, e.Name())
684	+			if err != nil {
685	+				continue
686	+			}
687	+			sessions = append(sessions, si)
688	+		}
689	+		if len(sessions) == 0 {
690	+			continue
691	+		}
692	+		sort.Slice(sessions, func(i, j int) bool {
693	+			return sessions[i].ModTime.After(sessions[j].ModTime)
694	+		})
695	+		groups = append(groups, ProjectGroup{
696	+			ProjectDir:  e.Name(),
697	+			ProjectPath: sessions[0].ProjectPath,
698	+			Sessions:    sessions,
699	+		})
700	+	}
701	+	sort.Slice(groups, func(i, j int) bool {
702	+		return groups[i].Sessions[0].ModTime.After(groups[j].Sessions[0].ModTime)
703	+	})
704	+	return groups, nil
705	+}

718	+// FindSession resolves a session id or unambiguous prefix across all projects.
719	+func FindSession(root, idOrPrefix string) (SessionInfo, error) {
720	+	groups, err := Scan(root)
721	+	if err != nil {
722	+		return SessionInfo{}, err
723	+	}
724	+	var matches []SessionInfo
725	+	for _, g := range groups {
726	+		for _, s := range g.Sessions {
727	+			if s.ID == idOrPrefix {
728	+				return s, nil // exact wins immediately
729	+			}
730	+			if strings.HasPrefix(s.ID, idOrPrefix) {
731	+				matches = append(matches, s)
732	+			}
733	+		}
734	+	}
735	+	switch len(matches) {
736	+	case 0:
737	+		return SessionInfo{}, fmt.Errorf("no session matching %q", idOrPrefix)
738	+	case 1:
739	+		return matches[0], nil
740	+	default:
741	+		return SessionInfo{}, fmt.Errorf("ambiguous prefix %q matches %d sessions", idOrPrefix, len(matches))
742	+	}
743	+}

1	# ccwhid (cc-what-have-i-done) — Design
2	
3	**Date:** 2026-07-11
4	**Status:** Approved
5	
6	## 1. Purpose & Scope
7	
8	A single, self-contained Go binary that turns a Claude Code session transcript
9	(`.jsonl`) into a browsable static HTML report styled like Claude's docs /
10	artifacts. Run it with flags to target a specific session, or with no args to
11	open a TUI session browser. Output is a self-contained folder that works over
12	`file://`, needs no server, and is safe to commit into a project repo as a
13	documentation artifact.
14	
15	The report answers: "what did I prompt, and what did Claude Code do?"
16	
17	## 2. Architecture (packages)
18	
19	```
20	cmd/ccwhid/main.go        wiring: cobra root command, flag parsing, dispatch
21	internal/discovery        find & decode ~/.claude/projects, list sessions + metadata
22	internal/transcript       parse JSONL into the typed domain model
23	internal/model            domain types: Session, Turn, Message, ToolCall, Diff, Subagent
24	internal/redact           secret/path redaction (pluggable rules)
25	internal/render           build the HTML site from the model (html/template + goldmark + chroma)
26	internal/assets           go:embed of templates, css, js
27	internal/tui              Bubble Tea session browser + options screen
28	```
29	
30	Boundaries: `transcript` produces a `model.Session` and is the only package that
31	knows the JSONL wire format. `render` consumes `model.Session` and knows nothing
32	about JSONL. `tui` orchestrates only — it calls `discovery` and `render`, and
33	contains no parsing or HTML logic. Each package is independently testable.
34	
35	**External libraries:** cobra (CLI), Bubble Tea + Bubbles + Lipgloss (TUI),
36	goldmark (markdown → HTML), chroma (syntax highlighting). All HTML/CSS/JS assets
37	embedded via `go:embed`. No external fonts or CDNs at runtime.
38	
39	## 3. Transcript Format (as observed)
40	
41	Sessions live at `~/.claude/projects/<encoded-cwd>/<session-id>.jsonl`. The
42	encoded directory name is the absolute cwd with `/` replaced by `-` (e.g.
43	`-Users-markus-IdeaProjects-cc-what-have-i-done`).
44	
45	Each line is a JSON object with a `type`:
46	- `user` / `assistant` — carry a `message` field in Anthropic message format
47	  (content blocks: `text`, `thinking`, `tool_use`, `tool_result`). Threaded via
48	  `parentUuid`/`uuid`. Metadata: `timestamp`, `cwd`, `gitBranch`, `slug`,
49	  `version`, `isSidechain`, `isMeta`, `userType`, `toolUseResult`.
50	- `attachment` — injected attachments.
51	- `ai-title` — a human-readable session title (`aiTitle`).
52	- `last-prompt`, `mode`, `permission-mode`, `file-history-snapshot` — session
53	  state records.
54	
55	**Subagents:** Task-tool subagent activity appears as records with
56	`isSidechain: true` in the *same* file, linked to the spawning tool call. These
57	are folded into the parent `Task` tool call as nested subagent runs.
58	
59	**Robustness:** the parser must tolerate unknown `type` values, unknown content
60	block kinds, and malformed lines (see §9).
61	
62	## 4. Data Flow
63	
64	1. `discovery` scans `~/.claude/projects/*/*.jsonl`, decodes project paths, and
