81	// with no selector flags.
82	package tui
83	
84	import (
85		"fmt"

87	
88		tea "github.com/charmbracelet/bubbletea"
89		"github.com/charmbracelet/lipgloss"

94	// Row is one line in the browser: either a project header or a session.
95	type Row struct {
96		IsHeader bool
97		Label    string
98		Session  discovery.SessionInfo
99	}
100	
101	// Selection is the result of running the browser.
102	type Selection struct {
103		Session          discovery.SessionInfo
104		IncludeSubagents bool
105		Redact           bool
106		Open             bool
107		OutDir           string
108		Canceled         bool
109	}
110	
111	func flattenRows(groups []discovery.ProjectGroup) []Row {
112		var rows []Row
113		for _, g := range groups {
114			rows = append(rows, Row{IsHeader: true, Label: g.ProjectPath})
115			for _, s := range g.Sessions {
116				rows = append(rows, Row{Session: s})
117			}
118		}
119		return rows
120	}
121	
122	func firstSelectable(rows []Row) int {
123		for i, r := range rows {
124			if !r.IsHeader {
125				return i
126			}
127		}
128		return 0
129	}
130	
131	var (
132		headerStyle   = lipgloss.NewStyle().Bold(true).Foreground(lipgloss.Color("#D97757")).MarginTop(1)
133		selectedStyle = lipgloss.NewStyle().Foreground(lipgloss.Color("#1A1A18")).Background(lipgloss.Color("#F6E7DF"))
134		mutedStyle    = lipgloss.NewStyle().Foreground(lipgloss.Color("240"))
135		optionStyle   = lipgloss.NewStyle().Foreground(lipgloss.Color("#D97757"))
136	)
137	
138	type screen int
139	
140	const (
141		screenList screen = iota
142		screenOptions
143	)
144	
145	type model struct {
146		rows      []Row
147		cursor    int
148		screen    screen
149		sel       Selection
150		optCursor int
151		done      bool
152	}
153	
154	func newModel(groups []discovery.ProjectGroup) model {
155		rows := flattenRows(groups)
156		return model{
157			rows:   rows,
158			cursor: firstSelectable(rows),
159			sel: Selection{
160				IncludeSubagents: true,
161				Redact:           true,
162			},
163		}
164	}
165	
166	func (m model) Init() tea.Cmd { return nil }
167	
168	func (m model) Update(msg tea.Msg) (tea.Model, tea.Cmd) {
169		key, ok := msg.(tea.KeyMsg)
170		if !ok {
171			return m, nil
172		}
173		switch m.screen {
174		case screenList:
175			return m.updateList(key)
176		default:
177			return m.updateOptions(key)
178		}
179	}
180	
181	func (m model) updateList(key tea.KeyMsg) (tea.Model, tea.Cmd) {
182		switch key.String() {
183		case "ctrl+c", "q", "esc":
184			m.sel.Canceled = true
185			m.done = true
186			return m, tea.Quit
187		case "up", "k":
188			m.moveCursor(-1)
189		case "down", "j":
190			m.moveCursor(1)
191		case "enter":
192			if !m.rows[m.cursor].IsHeader {
193				m.sel.Session = m.rows[m.cursor].Session
194				m.screen = screenOptions
195			}
196		}
197		return m, nil
198	}
199	
200	func (m *model) moveCursor(delta int) {
201		i := m.cursor
202		for {
203			i += delta
204			if i < 0 || i >= len(m.rows) {
205				return // out of bounds; keep current
206			}
207			if !m.rows[i].IsHeader {
208				m.cursor = i
209				return
210			}
211		}
212	}
213	
214	// optionCount is the number of toggle rows plus the Generate action.
215	const optionCount = 4 // subagents, redact, open, generate
216	
217	func (m model) updateOptions(key tea.KeyMsg) (tea.Model, tea.Cmd) {
218		switch key.String() {
219		case "ctrl+c", "q":
220			m.sel.Canceled = true
221			m.done = true
222			return m, tea.Quit
223		case "esc":
224			m.screen = screenList
225		case "up", "k":
226			if m.optCursor > 0 {
227				m.optCursor--
228			}
229		case "down", "j":
230			if m.optCursor < optionCount-1 {
231				m.optCursor++
232			}
233		case " ", "enter":
234			switch m.optCursor {
235			case 0:
236				m.sel.IncludeSubagents = !m.sel.IncludeSubagents
237			case 1:
238				m.sel.Redact = !m.sel.Redact
239			case 2:
240				m.sel.Open = !m.sel.Open
241			case 3:
242				m.done = true
243				return m, tea.Quit
244			}
245		}
246		return m, nil
247	}
248	
249	func (m model) View() string {
250		if m.screen == screenOptions {
251			return m.viewOptions()
252		}
253		return m.viewList()
254	}
255	
256	func (m model) viewList() string {
257		var b strings.Builder
258		b.WriteString("Select a session  ↑/↓ move · enter select · q quit\n")
259		for i, r := range m.rows {
260			if r.IsHeader {
261				b.WriteString(headerStyle.Render(r.Label) + "\n")
262				continue
263			}
264			line := fmt.Sprintf("  %s  %s", r.Session.DisplayLabel(), mutedStyle.Render(r.Session.ID[:min(8, len(r.Session.ID))]))
265			if i == m.cursor {
266				line = selectedStyle.Render("▸ " + strings.TrimLeft(line, " "))
267			}
268			b.WriteString(line + "\n")
269		}
270		return b.String()
271	}
272	
273	func (m model) viewOptions() string {
274		var b strings.Builder
275		b.WriteString("Options for: " + m.sel.Session.DisplayLabel() + "\n\n")
276		toggles := []struct {
277			label string
278			on    bool
279		}{
280			{"Include subagents", m.sel.IncludeSubagents},
281			{"Redact secrets", m.sel.Redact},
282			{"Open in browser when done", m.sel.Open},
283		}
284		for i, t := range toggles {
285			check := "[ ]"
286			if t.on {
287				check = "[x]"
288			}
289			line := fmt.Sprintf("%s %s", check, t.label)
290			if m.optCursor == i {
291				line = optionStyle.Render("▸ " + line)
292			} else {
293				line = "  " + line
294			}
295			b.WriteString(line + "\n")
296		}
297		gen := "Generate report"
298		if m.optCursor == 3 {
299			gen = optionStyle.Render("▸ " + gen)
300		} else {
301			gen = "  " + gen
302		}
303		b.WriteString("\n" + gen + "\n\n")
304		b.WriteString(mutedStyle.Render("space toggle · enter confirm · esc back · q quit") + "\n")
305		return b.String()
306	}
307	
308	func min(a, b int) int {
309		if a < b {
310			return a
311		}
312		return b
313	}
314	
315	// Run launches the browser and returns the user's selection.
316	func Run(groups []discovery.ProjectGroup) (Selection, error) {
317		if len(groups) == 0 {
318			return Selection{Canceled: true}, fmt.Errorf("no sessions found under ~/.claude/projects")
319		}
320		m := newModel(groups)
321		p := tea.NewProgram(m)
322		res, err := p.Run()
323		if err != nil {
324			return Selection{}, err
325		}
326		fm := res.(model)
327		return fm.sel, nil
328	}

// Scan indexes every session under root (the projects directory), grouped by
// project. Groups are ordered by most-recent session first; sessions within a
// group are newest-first.
func Scan(root string) ([]ProjectGroup, error) {
	entries, err := os.ReadDir(root)
	if err != nil {
		return nil, err
	}
	var groups []ProjectGroup
	for _, e := range entries {
		if !e.IsDir() {
			continue
		}
		dir := filepath.Join(root, e.Name())
		files, err := filepath.Glob(filepath.Join(dir, "*.jsonl"))
		if err != nil {
			return nil, err
		}
		var sessions []SessionInfo
		for _, fp := range files {
			si, err := indexFile(fp, e.Name())
			if err != nil {
				continue
			}
			sessions = append(sessions, si)
		}
		if len(sessions) == 0 {
			continue
		}
		sort.Slice(sessions, func(i, j int) bool {
			return sessions[i].ModTime.After(sessions[j].ModTime)
		})
		groups = append(groups, ProjectGroup{
			ProjectDir:  e.Name(),
			ProjectPath: sessions[0].ProjectPath,
			Sessions:    sessions,
		})
	}
	sort.Slice(groups, func(i, j int) bool {
		return groups[i].Sessions[0].ModTime.After(groups[j].Sessions[0].ModTime)
	})
	return groups, nil
}
