111	package usage
112	
113	import (
114		"sort"
115	
116		"github.com/saigyo/cc-what-have-i-done/internal/model"
117	)
118	
119	// TokenCounts is a bucket of token totals by billing category.
120	type TokenCounts struct {
121		Input        int
122		Output       int
123		CacheRead    int
124		CacheWrite5m int
125		CacheWrite1h int
126	}
127	
128	// InOut is the base input+output total, excluding cache tokens. Used for the
129	// report's headline figure so cache-read volume does not dominate it.
130	func (t TokenCounts) InOut() int { return t.Input + t.Output }
131	
132	func (t *TokenCounts) add(u model.Usage) {
133		t.Input += u.Input
134		t.Output += u.Output
135		t.CacheRead += u.CacheRead
136		t.CacheWrite5m += u.CacheWrite5m
137		t.CacheWrite1h += u.CacheWrite1h
138	}
139	
140	// ModelUsage is the token total and cost for one model in a session.
141	type ModelUsage struct {
142		Model   string
143		Tokens  TokenCounts
144		CostUSD *float64 // nil when the model is unpriced
145	}
146	
147	// Report is the computed usage for a session.
148	type Report struct {
149		Total           TokenCounts
150		TotalCostUSD    *float64        // nil only when nothing was priced
151		ByModel         []ModelUsage    // sorted, highest in+out first
152		PerTurnCost     map[int]*float64 // top-level turn index -> cost (nil = unpriced)
153		PricesAsOf      string
154		HasUnknownModel bool // some tokens came from an unpriced model
155		HasAnyUsage     bool // at least one turn carried usage
156	}
157	
158	// cost returns the USD cost of a token bucket at a model's price.
159	func cost(t TokenCounts, p Price) float64 {
160		inRate := p.Input / 1e6
161		return float64(t.Input)*inRate +
162			float64(t.Output)*(p.Output/1e6) +
163			float64(t.CacheWrite5m)*inRate*cacheWrite5mMult +
164			float64(t.CacheWrite1h)*inRate*cacheWrite1hMult +
165			float64(t.CacheRead)*inRate*cacheReadMult
166	}
167	
168	// Compute aggregates token usage and estimated cost for a session. Totals and
169	// per-model figures include nested subagent turns; per-turn costs are recorded
170	// for top-level turns only.
171	func Compute(s model.Session) Report {
172		r := Report{PricesAsOf: PricesAsOf, PerTurnCost: map[int]*float64{}}
173		byModel := map[string]*TokenCounts{}
174		var order []string
175	
176		accumulate := func(t model.Turn) {
177			if t.Usage == nil {
178				return
179			}
180			r.HasAnyUsage = true
181			r.Total.add(*t.Usage)
182			tc, ok := byModel[t.Model]
183			if !ok {
184				tc = &TokenCounts{}
185				byModel[t.Model] = tc
186				order = append(order, t.Model)
187			}
188			tc.add(*t.Usage)
189		}
190	
191		// Totals + per-model: walk top-level and nested subagent turns.
192		for _, t := range s.Turns {
193			walk(t, accumulate)
194		}
195	
196		// Per-turn cost for top-level turns only.
197		for i, t := range s.Turns {
198			if t.Usage == nil {
199				continue
200			}
201			if p, ok := Lookup(t.Model); ok {
202				c := cost(counts(*t.Usage), p)
203				r.PerTurnCost[i] = &c
204			} else {
205				r.PerTurnCost[i] = nil
206			}
207		}
208	
209		// Per-model rows + grand total cost.
210		var total float64
211		priced := false
212		for _, m := range order {
213			tc := *byModel[m]
214			mu := ModelUsage{Model: m, Tokens: tc}
215			if p, ok := Lookup(m); ok {
216				c := cost(tc, p)
217				mu.CostUSD = &c
218				total += c
219				priced = true
220			} else {
221				r.HasUnknownModel = true
222			}
223			r.ByModel = append(r.ByModel, mu)
224		}
225		if priced {
226			r.TotalCostUSD = &total
227		}
228		sort.SliceStable(r.ByModel, func(i, j int) bool {
229			return r.ByModel[i].Tokens.InOut() > r.ByModel[j].Tokens.InOut()
230		})
231		return r
232	}
233	
234	// counts converts a model.Usage into a TokenCounts.
235	func counts(u model.Usage) TokenCounts {
236		return TokenCounts{Input: u.Input, Output: u.Output, CacheRead: u.CacheRead,
237			CacheWrite5m: u.CacheWrite5m, CacheWrite1h: u.CacheWrite1h}
238	}
239	
240	// walk visits a turn and any turns nested in its Task subagents.
241	func walk(t model.Turn, fn func(model.Turn)) {
242		fn(t)
243		for _, b := range t.Blocks {
244			if b.Tool == nil {
245				continue
246			}
247			for _, sub := range b.Tool.Subagents {
248				for _, st := range sub.Turns {
249					walk(st, fn)
250				}
251			}
252		}
253	}
254	```
