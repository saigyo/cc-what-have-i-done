278	// clampScroll returns a scroll offset that keeps cursor within the visible
279	// window [top, top+visible) over a list of length n.
280	func clampScroll(cursor, top, visible, n int) int {
281		if visible < 1 {
282			visible = 1
283		}
284		if cursor < top {
285			top = cursor
286		}
287		if cursor >= top+visible {
288			top = cursor - visible + 1
289		}
290		if top < 0 {
291			top = 0
292		}
293		// Don't scroll past the end when the list shrank.
294		if max := n - visible; top > max && max >= 0 {
295			top = max
296		}
