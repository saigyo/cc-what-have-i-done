60	// Package model defines the domain types shared across ccwhid: a parsed
61	// Session and the timeline of Turns, Blocks and ToolCalls within it.
62	package model
63	
64	import "time"
65	
66	type TurnKind string
67	
68	const (
69		TurnUser      TurnKind = "user"
70		TurnAssistant TurnKind = "assistant"
71	)
72	
73	type BlockType string
74	
75	const (
76		BlockText     BlockType = "text"
77		BlockThinking BlockType = "thinking"
78		BlockToolUse  BlockType = "tool_use"
79	)
80	
81	// Session is one fully-parsed Claude Code transcript.
82	type Session struct {
83		ID           string
84		ProjectPath  string // decoded absolute cwd
85		Title        string
86		GitBranch    string
87		Version      string
88		StartedAt    time.Time
89		EndedAt      time.Time
90		Turns        []Turn
91		SkippedLines int // count of malformed lines skipped while parsing
92	}
93	
94	func (s Session) Duration() time.Duration { return s.EndedAt.Sub(s.StartedAt) }
95	
96	func (s Session) DisplayTitle() string {
97		if s.Title != "" {
98			return s.Title
99		}
100		return "Untitled session"
101	}
102	
103	// Turn is a single user or assistant message, holding ordered content blocks.
104	type Turn struct {
105		Kind      TurnKind
106		Timestamp time.Time
107		Blocks    []Block
108	}
109	
110	// Block is one content unit: assistant/user text, a thinking block, or a tool call.
111	type Block struct {
112		Type BlockType
113		Text string    // for BlockText and BlockThinking
114		Tool *ToolCall // for BlockToolUse
115	}
116	
117	// ToolCall is a single tool invocation and its result.
118	type ToolCall struct {
119		ID        string
120		Name      string // e.g. "Bash", "Edit", "Read", "Task"
121		Summary   string // one-line summary for the collapsed card header
122		InputJSON string // pretty-printed input for generic display
123		Result    *ToolResult
124		Diff      *Diff      // set for Edit/Write
125		Subagents []Subagent // set for Task calls with sidechain activity
126	}
127	
128	type ToolResult struct {
129		Content string
130		IsError bool
131	}
132	
133	type Diff struct {
134		Path    string
135		OldText string
136		NewText string
137	}
138	
139	// Subagent is a nested sidechain run attached to a parent Task tool call.
140	type Subagent struct {
141		Description string
142		Turns       []Turn
143	}
144	```
