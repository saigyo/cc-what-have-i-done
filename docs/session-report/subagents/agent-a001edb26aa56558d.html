174	import (
175		"encoding/json"

184	// agentMeta mirrors agent-<ID>.meta.json next to each agent transcript.
185	type agentMeta struct {
186		AgentType   string `json:"agentType"`
187		Description string `json:"description"`
188		ToolUseID   string `json:"toolUseId"`
189		SpawnDepth  int    `json:"spawnDepth"`
190	}
191	
192	// LoadAgentSessions parses the linked subagent transcripts of a root session:
193	// <transcriptPath minus .jsonl>/subagents/agent-*.jsonl, sorted by file name.
194	// A missing subagents dir yields (nil, nil); an unparsable agent file is
195	// skipped with a warning on stderr. Nested agents (spawnDepth > 1) live in the
196	// same flat directory and are returned alongside depth-1 agents.
197	func LoadAgentSessions(transcriptPath string, opts Options) ([]model.AgentSession, error) {
198		dir := filepath.Join(strings.TrimSuffix(transcriptPath, ".jsonl"), "subagents")
199		entries, err := os.ReadDir(dir)
200		if err != nil {
201			if os.IsNotExist(err) {
202				return nil, nil
203			}
204			return nil, err
205		}
206		opts.AgentFile = true
207		var out []model.AgentSession
208		for _, e := range entries { // ReadDir sorts by name
209			name := e.Name()
210			if e.IsDir() || !strings.HasPrefix(name, "agent-") || !strings.HasSuffix(name, ".jsonl") {
211				continue
212			}
213			id := strings.TrimSuffix(strings.TrimPrefix(name, "agent-"), ".jsonl")
214			sess, err := ParseFile(filepath.Join(dir, name), opts)
215			if err != nil {
216				fmt.Fprintf(os.Stderr, "warning: skipping agent transcript %s: %v\n", name, err)
217				continue
218			}
219			a := model.AgentSession{ID: id, Description: id, Session: sess}
220			if b, err := os.ReadFile(filepath.Join(dir, "agent-"+id+".meta.json")); err == nil {
221				var m agentMeta
222				if json.Unmarshal(b, &m) == nil {
223					if m.Description != "" {
224						a.Description = m.Description
225					}
226					a.AgentType = m.AgentType
227					a.ToolUseID = m.ToolUseID
228					a.SpawnDepth = m.SpawnDepth
229				}
230			}
231			out = append(out, a)
232		}
233		return out, nil
234	}
235	```
