176	// parseTaskNotification extracts fields from a <task-notification> payload with
177	// tolerant string scanning (the payload is pseudo-XML with unescaped markdown
178	// inside <result>). ok is false unless both task-id and summary are present.
179	func parseTaskNotification(s string) (taskNotification, bool) {
180		t := trimLeftSpace(s)
181		if !hasPrefix(t, "<task-notification>") {
182			return taskNotification{}, false
183		}
184		n := taskNotification{
185			TaskID:    tagContent(t, "task-id"),
186			ToolUseID: tagContent(t, "tool-use-id"),
187			Status:    tagContent(t, "status"),
188			Summary:   tagContent(t, "summary"),
189			Result:    tagContent(t, "result"),
190		}
191		if n.TaskID == "" || n.Summary == "" {
192			return taskNotification{}, false
193		}
194		return n, true
195	}

197	// tagContent returns the text between the first <name> and the last </name>,
198	// trimmed. The last closing tag guards <result> bodies that themselves contain
199	// XML-looking text.
200	func tagContent(s, name string) string {
201		open, close := "<"+name+">", "</"+name+">"
202		i := strings.Index(s, open)
203		if i < 0 {
204			return ""
205		}
206		rest := s[i+len(open):]
207		j := strings.LastIndex(rest, close)
208		if j < 0 {
209			return ""
210		}
211		return strings.TrimSpace(rest[:j])
212	}
