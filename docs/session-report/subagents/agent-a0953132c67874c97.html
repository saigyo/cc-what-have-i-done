169		entries, err := os.ReadDir(root)

171			return nil, err
172		}
173		var groups []ProjectGroup
174		for _, e := range entries {
175			if !e.IsDir() {
176				continue
177			}
178			dir := filepath.Join(root, e.Name())

180			if err != nil {
181				return nil, err
182			}
183			var sessions []SessionInfo
184			for _, fp := range files {
185				si, err := indexFile(fp, e.Name())
186				if err != nil {
187					continue
188				}
189				sessions = append(sessions, si)
190			}
191			if len(sessions) == 0 {
192				continue
193			}
194			sort.Slice(sessions, func(i, j int) bool {
195				return sessions[i].ModTime.After(sessions[j].ModTime)
196			})
197			groups = append(groups, ProjectGroup{
198				ProjectDir:  e.Name(),
199				ProjectPath: sessions[0].ProjectPath,
200				Sessions:    sessions,
201			})
202		}
203		sort.Slice(groups, func(i, j int) bool {
204			return groups[i].Sessions[0].ModTime.After(groups[j].Sessions[0].ModTime)
205		})
206		return groups, nil
207	}

210	func FindSession(root, idOrPrefix string) (SessionInfo, error) {

212		if err != nil {
213			return SessionInfo{}, err
214		}

216		for _, g := range groups {
217			for _, s := range g.Sessions {
218				if s.ID == idOrPrefix {
219					return s, nil // exact wins immediately
220				}

222					matches = append(matches, s)
223				}
224			}
225		}

227		case 0:

229		case 1:
230			return matches[0], nil
231		default:

31	// Highlight renders a code string with chroma using inline styles (no external
32	// stylesheet needed). lang may be empty for auto-detection.
33	func Highlight(code, lang string) template.HTML {
34		lexer := lexers.Get(lang)
35		if lexer == nil {
36			lexer = lexers.Analyse(code)
37		}
38		if lexer == nil {
39			lexer = lexers.Fallback
40		}
41		style := styles.Get("github")
42		if style == nil {
43			style = styles.Fallback
44		}
45		formatter := chromahtml.New(chromahtml.WithClasses(false), chromahtml.TabWidth(2))
46		it, err := lexer.Tokenise(nil, code)
47		if err != nil {
48			return template.HTML("<pre>" + html.EscapeString(code) + "</pre>")
49		}
50		var buf bytes.Buffer
51		if err := formatter.Format(&buf, style, it); err != nil {
52			return template.HTML("<pre>" + html.EscapeString(code) + "</pre>")
53		}
54		return template.HTML(buf.String())
55	}
