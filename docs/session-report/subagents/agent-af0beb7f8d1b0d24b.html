274	// clampScroll returns a scroll offset that keeps cursor within the visible
275	// window [top, top+visible) over a list of length n.
276	func clampScroll(cursor, top, visible, n int) int {
277		if visible < 1 {
278			visible = 1
279		}
280		if cursor < top {
281			top = cursor
282		}
283		if cursor >= top+visible {
284			top = cursor - visible + 1
285		}
286		if top < 0 {
287			top = 0
288		}
289		// Don't scroll past the end when the list shrank.
290		if max := n - visible; top > max && max >= 0 {
291			top = max
292		}
293		return top
294	}
