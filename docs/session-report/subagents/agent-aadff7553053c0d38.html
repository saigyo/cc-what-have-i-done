80	// rawRecord is one decoded line of a session .jsonl file.
81	type rawRecord struct {
82		Type        string          `json:"type"`
83		UUID        string          `json:"uuid"`
84		ParentUUID  string          `json:"parentUuid"`
85		Timestamp   string          `json:"timestamp"`
86		GitBranch   string          `json:"gitBranch"`
87		Version     string          `json:"version"`
88		Cwd         string          `json:"cwd"`
89		AiTitle     string          `json:"aiTitle"`
90		IsSidechain bool            `json:"isSidechain"`
91		IsMeta      bool            `json:"isMeta"`
92		Message     json.RawMessage `json:"message"`
93	}
94	
95	// apiBlock is one content block of an Anthropic message.
96	type apiBlock struct {
97		Type      string          `json:"type"`
98		Text      string          `json:"text"`
99		Thinking  string          `json:"thinking"`
100		ID        string          `json:"id"`
101		Name      string          `json:"name"`
102		Input     json.RawMessage `json:"input"`
103		ToolUseID string          `json:"tool_use_id"`
104		Content   json.RawMessage `json:"content"`
105		IsError   bool            `json:"is_error"`
106	}
107	
108	// decodeMessageContent extracts the role and content blocks from a raw message
109	// object. Content may be a plain string (wrapped as a single text block) or an
110	// array of blocks.
111	func decodeMessageContent(raw json.RawMessage) (string, []apiBlock) {
112		if len(raw) == 0 {
113			return "", nil
114		}
115		var msg struct {
116			Role    string          `json:"role"`
117			Content json.RawMessage `json:"content"`
118		}
119		if err := json.Unmarshal(raw, &msg); err != nil {
120			return "", nil
121		}
122		var s string
123		if json.Unmarshal(msg.Content, &s) == nil {
124			return msg.Role, []apiBlock{{Type: "text", Text: s}}
125		}
126		var blocks []apiBlock
127		_ = json.Unmarshal(msg.Content, &blocks) // tolerate; empty on failure
128		return msg.Role, blocks
129	}
130	
131	// toolResultText flattens a tool_result content payload to plain text.
132	func toolResultText(raw json.RawMessage) string {
133		if len(raw) == 0 {
134			return ""
135		}
136		var s string
137		if json.Unmarshal(raw, &s) == nil {
138			return s
139		}
140		var blocks []struct {
141			Type string `json:"type"`
142			Text string `json:"text"`
143		}
144		if json.Unmarshal(raw, &blocks) == nil {
145			var parts []string
146			for _, b := range blocks {
147				if b.Text != "" {
148					parts = append(parts, b.Text)
149				}
150			}
151			return strings.Join(parts, "\n")
152		}
153		return ""
154	}
155	
156	func parseTime(s string) time.Time {
157		if s == "" {
158			return time.Time{}
159		}
160		t, err := time.Parse(time.RFC3339, s)
161		if err != nil {
162			return time.Time{}
163		}
164		return t
165	}
166	```
